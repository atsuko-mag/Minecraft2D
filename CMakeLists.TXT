cmake_minimum_required(VERSION 3.13 FATAL_ERROR)

set(PROJECT_NAME Minecraft2D)

project(${PROJECT_NAME})

add_executable(${PROJECT_NAME} src/main.cpp src/Config.cpp src/Config.h src/Physics.h src/World.h src/Graphics.h src/Graphics.cpp src/CallBackFunctions.h src/CallBackFunctions.cpp)

set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(GLFW_INSTALL OFF CACHE BOOL "" FORCE)

add_subdirectory(external/glfw)
target_link_libraries(${PROJECT_NAME} glfw)

add_subdirectory(external/glad)
target_link_libraries(${PROJECT_NAME} glad)
